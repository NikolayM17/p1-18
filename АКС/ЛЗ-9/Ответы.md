### Задача 3
```x86asm
power:
    push %ebx
    mov $1, %ebx
    
loop:
    test %edx,%edx
    jz recovery
    imul %eax,%ebx
    dec %edx
    jnz loop

recovery:
    mov %ebx,%eax
    pop %ebx

ret
```

#Описать рекурсивную функцию DigitSum(K) целого типа, 
#которая находит сумму цифр целого числа K, не используя оператор цикла. 
#С помощью этой функции найти суммы цифр для пяти данных целых чисел.

def digitSum(k):
  k = abs(k)
  if k < 10:
    return k
  else:
    return k % 10 + digitSum((k - (k % 10))/10)

for i in range(5):
  print(digitSum(int(input())), end = ' ')

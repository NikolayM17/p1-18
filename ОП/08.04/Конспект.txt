Timer timePlus(Timer a, struct Time b){ 
    Timer tmp;
    int sec, min, h;
    sec = a.sec + b.sec;
    min = sec / 60 + a.min + b.min;
    h = min / 60 + a.hour + b.hour;
    tmp.sec = sec % 60;
    tmp.min = min % 60;
    tmp.hour = h % 12;
    return tmp;
};
